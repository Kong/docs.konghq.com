{% navtab Ingress %}
```bash
echo "
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: {{ name }}
  annotations:
    konghq.com/strip-path: 'true'
spec:
  ingressClassName: kong
  rules:
  - host: {{ hostname }}
    http:
      paths:
      - path: {{ path }}
        pathType: ImplementationSpecific
        backend:
          service:
            name: {{ service }}
            port:
              number: 1027
" | kubectl apply -f -
```
Response:
```text
ingress.networking.k8s.io/{{ name }} created
```
